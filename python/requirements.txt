# ── Core (always required) ────────────────────────────────────────────────────

# Fetches YouTube captions without the YouTube Data API.
# Also pulls in: requests, defusedxml
youtube-transcript-api>=1.0.0

# OpenAI-compatible client used for:
#   - LLM summarisation / dialogue generation (chat completions)
#   - TTS audio generation when --tts-engine openai (the default)
# Also pulls in: httpx, pydantic, anyio, tqdm, and others
openai>=1.30.0

# Used directly for cookie-based YouTube authentication (--cookies flag).
# Already a transitive dep of youtube-transcript-api, listed here because
# the script imports it directly.
requests>=2.28.0


# ── Optional: Kokoro local TTS (--tts-engine kokoro) ─────────────────────────
#
# Open-source TTS engine that runs on CPU — no API key needed.
# Model weights (~82 MB) are downloaded automatically from HuggingFace on
# first run and cached locally.
# Also pulls in: torch (CPU build), numpy, misaki, huggingface-hub, loguru
#
# Install with:
#   pip install kokoro
#
# kokoro>=0.9.0


# ── Optional: Frame capture (--num-frames N) ──────────────────────────────────
#
# Resolves a direct video stream URL from YouTube without downloading the file.
yt-dlp>=2024.1.0

# Seeks to timestamps in the video stream and saves JPEG frames.
# Use the headless build on servers (no GUI dependencies).
opencv-python-headless>=4.8.0


# ── Optional: HTML page generation (enabled by default, --no-page to skip) ───
#
# Converts the summary markdown to HTML inside the generated page.
# Without this, the markdown is shown as escaped plain text (still readable).
markdown>=3.5


# ── Notes ─────────────────────────────────────────────────────────────────────
#
# numpy  — only needed for Kokoro; arrives automatically as a kokoro dependency.
# ffmpeg — NOT required. OpenAI audio is concatenated as raw MP3 bytes;
#           Kokoro audio is written as WAV using Python's stdlib wave module.
# yt-dlp + opencv-python-headless — only needed for --num-frames (default: 5).
#           Set --num-frames 0 to skip frame capture entirely without needing them.
# markdown — only needed for the HTML page; falls back to escaped plain text
#             if not installed. Install with: pip install markdown
